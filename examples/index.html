standalone JS snippets

These show how to use webml-kit from code, without the UI. Run these in a project with npm install webml-kit @huggingface/transformers.

snippets/detect-device.js 
// Check what the user's machine can handle
import { detectDevice, canRun, formatSize } from 'webml-kit';

const device = await detectDevice();
console.log('backend:', device.backend);
console.log('dtype:', device.recommendedDtype);

if (device.gpu) {
  console.log('gpu:', device.gpu.vendor, device.gpu.architecture);
  console.log('vram:', formatSize(device.gpu.vram));
}

const check = await canRun('4GB');
console.log('can run 4GB model?', check.ok, check.reason ?? '');
snippets/check-cache.js 
// See what's already downloaded
import { isCached, listCachedModels, getCacheSize, formatSize } from 'webml-kit';

if (await isCached('onnx-community/Bonsai-1.7B-ONNX')) {
  console.log('model is cached, skip the progress bar');
}

const models = await listCachedModels();
for (const m of models) {
  console.log(`${m.modelId}: ${m.size}`);
}

console.log('total cache:', formatSize(await getCacheSize()));
snippets/text-generation.js 
// Stream text generation
import { ModelClient } from 'webml-kit';

const client = new ModelClient(
  new URL('webml-kit/worker', import.meta.url)
);

await client.load({
  task: 'text-generation',
  modelId: 'onnx-community/Bonsai-1.7B-ONNX',
  dtype: 'q4',
  onProgress: ({ percent }) => console.log(`loading: ${percent}%`),
});

for await (const { token, tps } of client.stream('Explain quantum computing')) {
  process.stdout.write(token);
}
// tps: ~40 tok/s on M-series Mac
snippets/image-classification.js 
// Classify an image
import { ModelClient } from 'webml-kit';

const client = new ModelClient(
  new URL('webml-kit/worker', import.meta.url)
);

await client.load({
  task: 'image-classification',
  modelId: 'Xenova/vit-base-patch16-224',
  dtype: 'fp32',
});

const labels = await client.run('image-classification', '/path/to/cat.jpg');
console.log(labels);
// [{ label: 'tabby cat', score: 0.98 }, ...]
snippets/multi-model.js 
// Load multiple models at once
import { ModelClient } from 'webml-kit';

const client = new ModelClient(
  new URL('webml-kit/worker', import.meta.url)
);

// Load text generation and embeddings side by side
await client.load({ task: 'text-generation', modelId: 'onnx-community/Bonsai-1.7B-ONNX', dtype: 'q4' });
await client.load({ task: 'feature-extraction', modelId: 'Xenova/all-MiniLM-L6-v2', dtype: 'fp32' });

// Use them independently
const embedding = await client.run('feature-extraction', 'Hello world');
console.log('embedding dims:', embedding.length);

for await (const { token } of client.stream('What is an embedding?')) {
  process.stdout.write(token);
}

// Free memory when done
client.dispose();
